validators/isInArr.js

import {
  curry,
} from 'ramda'

/**
  * Функция проверяет есть ли подобный элемент в массиве.
  * @category validators
  * @method
  * @since v0.1.0
  * @param {(*)} key - проверяемая перменная
  * @return {boolean}
  *
  * @example
  *
  *      isArr(12);      //=> false
  *      isArr([]);      //=> true
 */

export const isInArr = curry((el, arr) => arr.findIndex((arrEl) => arrEl === el) >= 0);
